Условные конструкции

Условные операторы позволяют выполнять различные действия в зависимости от условий.

Пример if/else

// Проверка возраста let age = 18; if (age < 18) { console.log("Вы несовершеннолетний"); } else if (age >= 18 && age < 65) { console.log("Вы взрослый"); } else { console.log("Вы пенсионер"); }

Проверка числа

Введите число, и мы определим его характеристику:

Тернарный оператор

// Синтаксис: условие ? значение_если_истина : значение_если_ложь let isMember = true; let discount = isMember ? 10 : 0; console.log(`Ваша скидка: ${discount}%`);

Switch/case

// Определение дня недели let day = 2; let dayName; switch(day) { case 1: dayName = "Понедельник"; break; case 2: dayName = "Вторник"; break; case 3: dayName = "Среда"; break; default: dayName = "Другой день"; } console.log(dayName); // Выведет "Вторник"

Калькулятор с выбором операции

Реализация калькулятора с использованием switch/case для выбора операции.

Введите данные для расчета

Код калькулятора

function calculate(a, b, operation) { let result; switch(operation) { case '+': result = a + b; break; case '-': result = a - b; break; case '*': result = a * b; break; case '/': result = b !== 0 ? a / b : "Ошибка: деление на ноль"; break; default: result = "Неизвестная операция"; } return result; }

Циклы в программировании

Циклы позволяют выполнять один и тот же код несколько раз.

Цикл for (счетчик)

// Синтаксис: for(инициализация; условие; инкремент) { ... } for(let i = 1; i <= 5; i++) { console.log("Итерация номер: " + i); }

Цикл while (условие)

// Синтаксис: while(условие) { ... } let count = 1; while(count <= 5) { console.log("Счет: " + count); count++; }

Цикл do...while (постусловие)

// Синтаксис: do { ... } while(условие) let num = 1; do { console.log("Число: " + num); num++; } while(num <= 5);

Управление циклами: break и continue

// break - выход из цикла for(let i = 1; i <= 10; i++) { if(i === 5) { break; // выходим из цикла при i = 5 } console.log(i); // выведет 1, 2, 3, 4 } // continue - пропуск итерации for(let i = 1; i <= 5; i++) { if(i === 3) { continue; // пропускаем итерацию при i = 3 } console.log(i); // выведет 1, 2, 4, 5 }

let secretNumber = Math.floor(Math.random() * 10) + 1; let attempts = 3; while(attempts > 0) { let userGuess = parseInt(prompt("Угадайте число (1-10):")); if(userGuess === secretNumber) { console.log("Поздравляем!
